package com.sap.netweaver.porta.mon;

import java.io.PrintStream;

public class ExitCommand extends Command {

	@Override
	public String execute() {
		PrintStream out = PortaMon.getOut();
		out.println();
		out.println("Exiting by user request...");
		System.exit(0);
		return null;
	}

}
