import de.outstare.fortbattleplayer.model.Area;
import de.outstare.fortbattleplayer.model.Combatant;

/**
 * the target of a combatant changed
 * 
 * @author daniel
 */
public class TargetAction extends CombatantAction {
	private final Area target;

	/**
	 * @param player
	 * @param target
	 */
	public TargetAction(final Combatant player, final Area target) {
		super(player);
		this.target = target;
	}

	/**
	 * @see de.outstare.fortbattleplayer.player.CombatantTurn#execute()
	 */
	public void execute() {
		player.setDestination(target);
	}

	/**
	 * @see java.lang.Object#toString()
	 */
	@Override
	public String toString() {
		return player + " moves to " + target;
	}
}
